pipeline {
    agent { docker { image 'maven:3.3.3' } }
    tools {
            maven 'Maven 3.3.9'
            jdk 'jdk8'
    }
    stages {
        stage('compile') {
          steps {
            echo 'Compiling....'
            sh 'mvn clean install'
          }
        }
        stage('build') {
            steps {
                echo 'Compiling....'
                sh 'mvn --version'
            }
        }
    }
}
